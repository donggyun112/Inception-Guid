     context: /*Dockerfile 경로*/ # Dockerfile 위치 지정
   ports:
     - "/*호스트 포트*/:/*컨테이너 포트*/" # 호스트와 컨테이너의 포트 매핑

   container_name: /*컨테이너 이름*/ # 컨테이너 이름 지정
